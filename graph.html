import matplotlib.pyplot as plt from mpl_toolkits.mplot3d import Axes3D import numpy as np import mplcursors def equation(x, y): return x**2 + y**2 x_values = np.linspace(-10, 10, 100) y_values = np.linspace(-10, 10, 100) X, Y = np.meshgrid(x_values, y_values) Z = equation(X, Y) fig = plt.figure() ax = fig.add_subplot(111, projection='3d') fig,ax.plot_surface(X, Y, Z, cmap='viridis', alpha=0.7) ax.set_xlabel('X-axis') ax.set_ylabel('Y-axis') ax.set_zlabel('Z-axis') mplcursors.cursor(hover=True).connect("add", lambda sel:sel.annotation.set_text(f"({sel.target[0]:.2f}, {sel.target[1]:.2f},{equation(sel.target[0], sel.target[1]):.2f})")) display(fig)
